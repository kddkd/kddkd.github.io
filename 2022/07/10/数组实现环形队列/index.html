数组实现环形队列

发表于 2022-07-10 
package queue;
    /*
        思路分析：
            1.使用环形数组模拟队列，可以使队列复用；
            2.front:指向队首位置，front = 0;
            3.rear:指向队尾的下一个位置，用于预留一个空位,rear = 0;
            4.队列为空：front = rear ;
            5.队列为满：(rear + 1) % maxSize = front;
            6.数据的有效个数：(rear + maxSize - front) % maxSize;
     */



import java.util.Scanner;

public class CircleArrayQueue {
public static void main(String[] args) {
    CircleQueue  queue = new CircleQueue(4);
    Boolean loop = true;
    char key = ' ';//用于接受用户输入字符
    Scanner scanner = new Scanner(System.in);

    while(loop){
            System.out.println("s(show):显示队列");
            System.out.println("a(add):添加数据到队列");
            System.out.println("g(get):获取队列数据");
            System.out.println("h(head):显示队列头数据");
            System.out.println("e(exit):退出");
            key = scanner.next().charAt(0);

            switch (key){
                case 's':
                    queue.showQueue();
                    break;
                case 'a':
                    int next = scanner.nextInt();
                    queue.addQueue(next);
                    break;
                case 'g':
                    try {
                        int res = queue.getQueue();
                        System.out.println("取出的数据为: " + res);
                    }catch (Exception e){
                        System.out.println(e.getMessage());
                    }
                    break;
                case 'h':
                    try {
                        int res = queue.headQueue();
                        System.out.println("取出的数据为: " + res);
                    }catch (Exception e){
                        System.out.println(e.getMessage());
                    }
                    break;
                case 'e':
                    loop = false;
                    System.out.println("退出程序...");
                    break;
            }
        }
    }
}


class CircleQueue{
    private int maxSize;//队列容量
    int[] arr;//用这个数组实现队列；
    int front;//指向头部位置的数据；
    int rear;//指向尾部数据的下一个

    public CircleQueue(int arrMaxSize){
        maxSize = arrMaxSize;
        arr = new int[maxSize];
        front = 0;
        rear = 0;
}


public boolean isEmpty(){
    return  front == rear;
}

public boolean isFull(){
    return (rear + 1) % maxSize == front;
}


//添加元素
public void addQueue(int n){
        if(isFull()){
            System.out.println("队列已经满了。。");
            return;
        }
        arr[rear] = n;
        rear = (rear + 1) % maxSize;
    }

    //取出数据
    public int getQueue(){
        if (isEmpty()){
            throw new RuntimeException("队列为空，无法取出数据~~");
        }
        int value = arr[front];
        front = (front + 1) % maxSize;
        return value;
    }
    //显式头部数据
    public int headQueue(){
        if (isEmpty()){
            throw new RuntimeException("队列为空，无法取出数据~~");
        }
        return arr[front];
    }

    //显式队列数据
    public void showQueue(){
        if (isEmpty()){
            System.out.println("队列为空..");
            return;
        }

        for (int i = front; i < front + size();i ++){
            System.out.printf("arr[%d] = %d\n",i % maxSize, arr[i % maxSize]);
        }
    }

    //数组有效个数
    public int size(){
        return (rear + maxSize - front) % maxSize;
    }
}

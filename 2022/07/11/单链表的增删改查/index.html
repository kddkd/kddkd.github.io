/*
    创建单链表：
        1.添加元素；
        2.打印链表；
 */
class SingleList{
    //创建一个头节点，不存放数据
    HeroNode head = new HeroNode(0,"","");

    //添加链表
    public void add(HeroNode heroNode){
        HeroNode temp = head;
        //找到链表末尾，接上去即可
        while(true){
            if(temp.next == null){
                break;
            }
            temp = temp.next;
        }
        temp.next = heroNode;
    }


    //根据 no 顺序添加到链表
    //如果no存在就提示信息...
    public void addByOrder(HeroNode heroNode){
        //先找到插入位置的前一个才能插入
        HeroNode temp = head;
        boolean flag = false;
        while(true){
            if(temp.next == null){//到了链表的最后一个
                break;
            }
            if(temp.next.no > heroNode.no){//到了插入位置
                break;
            }else if(temp.next.no == heroNode.no){//no相同
                flag = true;
                break;
            }
            temp = temp.next;
        }
        if(flag){
            System.out.println("插入节点的编号相同，无法插入");
        }else {
            heroNode.next = temp.next;
            temp.next = heroNode;
        }
    }

    //修改元素信息，根据no修改
    public void update(HeroNode heroNode){
        if(head.next == null){
            System.out.println("链表为空，无法修改...");
            return;
        }
        boolean flag = false;
        HeroNode temp = head.next;
        while(true){
            if(temp.no == heroNode.no){
                flag = true;
                break;
            }

            if (temp.next == null){
                break;//没找到相同no
            }
            temp = temp.next;
        }

        if (flag){
            temp.name = heroNode.name;
            temp.nickName = heroNode.nickName;
        }else {
            System.out.println("链表无此节点。。。");
        }
    }
    //根据no删除节点
    //没有这个节点就提示
    public void delete(int no){
        //找删除结点的前一个结点 temp.next = heroNode.next即可
        if (head.next == null){
            System.out.println("空链表，无法删除节点...");
        }

        HeroNode temp = head;
        boolean flag = false;

        while(true){

            if(temp.next == null){//到了链表末尾
                break;
            }

            if (temp.next.no == no){
                flag = true;
                break;
            }

            temp = temp.next;
        }

        if (flag){
            temp.next = temp.next.next;
        }else {
            System.out.println("节点不存在...");
        }
    }
    //显式链表
    public void showList(){

        if(head.next == null){
            System.out.println("链表为空...");
        }
        HeroNode temp = head.next;
        //遍历显示temp元素
        while(true){
            if (temp == null){
                break;
            }

            System.out.println(temp);
            temp = temp.next;
        }
    }
}


class HeroNode{
    int no;//编号
    String name;//姓名
    String nickName;//外号
    HeroNode next;//指向下一个

    public HeroNode(int hNo,String hName,String hNickName){
        this.no = hNo;
        this.name = hName;
        this.nickName = hNickName;
    }

    @Override
    public String toString() {
        return "HeroNode{" +
                "no=" + no +
                ", name='" + name + '\'' +
                ", nickName='" + nickName + '\'' +
                '}';
    }
}

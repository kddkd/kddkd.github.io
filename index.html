     * 双向链表的基础操作：增删改查

    public class DoubleLinkedList {
        public static void main(String[] args) {
            HeroNode2 heroNode1 = new HeroNode2(1,"宋江","及时雨");
            HeroNode2 heroNode2 = new HeroNode2(2,"吴用","智多星");
            HeroNode2 heroNode3 = new HeroNode2(3,"卢俊义","玉麒麟");
            HeroNode2 heroNode4 = new HeroNode2(4,"林冲","豹子头");

            DoubleList doubleList = new DoubleList();
            doubleList.add(heroNode1);
            doubleList.add(heroNode2);
            doubleList.add(heroNode3);
            doubleList.add(heroNode4);

            System.out.println("--------------");
            doubleList.update(heroNode5);

            doubleList.showList();
            System.out.println("--------------");

    class DoubleList{
        //创建一个头节点，不存放数据
        private HeroNode2 head = new HeroNode2(0,"","");

        public HeroNode2 getHead() {
            return head;
        }

        //添加链表
        public void add(HeroNode2 heroNode){
            HeroNode2 temp = head;
            //找到链表末尾，接上去即可
            while(true){
                if(temp.next == null){
                    break;
                }
                temp = temp.next;
            }
            temp.next = heroNode;
            heroNode.pre = temp;//双向链表的pre需要指回去
        }

        //修改元素信息，根据no修改
        public void update(HeroNode2 heroNode){
            if(head.next == null){
                System.out.println("链表为空，无法修改...");
                return;
            }
            boolean flag = false;
            HeroNode2 temp = head.next;
            while(true){
                if(temp.no == heroNode.no){
                    flag = true;
                    break;
                }

                if (temp.next == null){
                    break;//没找到相同no
                }
                temp = temp.next;
            }

            if (flag){
                temp.name = heroNode.name;
                temp.nickName = heroNode.nickName;
            }else {
                System.out.println("链表无此节点。。。");
            }
        }


        //根据no删除节点
        //没有这个节点就提示
        public void delete(int no){
            //对于双向链表，我可以直接找到需要删除的节点
            if (head.next == null){
                System.out.println("空链表，无法删除节点...");
            }

            HeroNode2 temp = head.next;
            boolean flag = false;

            while(true){
                if (temp == null){
                    break;//遍历到末尾没找到
                }

                if(temp.no == no){
                    flag = true;
                    break;
                }

                temp = temp.next;

            }

            if (flag){
                //删除
                temp.pre.next = temp.next;
                if(temp.next != null){
                    temp.next.pre = temp.pre;
                }
            }else {
                System.out.println("当前节点不存在...");
            }
        }

        //显式链表
        public void showList(){

            if(head.next == null){
                System.out.println("链表为空...");
            }
            HeroNode2 temp = head.next;
            //遍历显示temp元素
            while(true){
                if (temp == null){
                    break;
                }

                System.out.println(temp);
                temp = temp.next;
            }
        }
    }

    class HeroNode2{
        int no;//编号
        String name;//姓名
        String nickName;//外号
        HeroNode2 next;//指向下一个
        HeroNode2 pre;

        public HeroNode2(int hNo,String hName,String hNickName){
            this.no = hNo;
            this.name = hName;
            this.nickName = hNickName;
        }

        @Override
        public String toString() {
            return "HeroNode{" +
                    "no=" + no +
                    ", name='" + name + '\'' +
                    ", nickName='" + nickName + '\'' +
                    '}';
        }
    }

package queue;
    /*
        思路分析：
            1.使用环形数组模拟队列，可以使队列复用；
            2.front:指向队首位置，front = 0;
            3.rear:指向队尾的下一个位置，用于预留一个空位,rear = 0;
            4.队列为空：front = rear ;
            5.队列为满：(rear + 1) % maxSize = front;
            6.数据的有效个数：(rear + maxSize - front) % maxSize;
     */



import java.util.Scanner;

public class CircleArrayQueue {
public static void main(String[] args) {
    CircleQueue  queue = new CircleQueue(4);
    Boolean loop = true;
    char key = ' ';//用于接受用户输入字符
    Scanner scanner = new Scanner(System.in);

    while(loop){
            System.out.println("s(show):显示队列");
            System.out.println("a(add):添加数据到队列");
            System.out.println("g(get):获取队列数据");
            System.out.println("h(head):显示队列头数据");
            System.out.println("e(exit):退出");
            key = scanner.next().charAt(0);

            switch (key){
                case 's':
                    queue.showQueue();
                    break;
                case 'a':
                    int next = scanner.nextInt();
                    queue.addQueue(next);
                    break;
                case 'g':
                    try {
                        int res = queue.getQueue();
                        System.out.println("取出的数据为: " + res);
                    }catch (Exception e){
                        System.out.println(e.getMessage());
                    }
                    break;
                case 'h':
                    try {
                        int res = queue.headQueue();
                        System.out.println("取出的数据为: " + res);
                    }catch (Exception e){
                        System.out.println(e.getMessage());
                    }
                    break;
                case 'e':
                    loop = false;
                    System.out.println("退出程序...");
                    break;
            }
        }
    }
}


class CircleQueue{
    private int maxSize;//队列容量
    int[] arr;//用这个数组实现队列；
    int front;//指向头部位置的数据；
    int rear;//指向尾部数据的下一个

    public CircleQueue(int arrMaxSize){
        maxSize = arrMaxSize;
        arr = new int[maxSize];
        front = 0;
        rear = 0;
}


public boolean isEmpty(){
    return  front == rear;
}

public boolean isFull(){
    return (rear + 1) % maxSize == front;
}


//添加元素
public void addQueue(int n){
        if(isFull()){
            System.out.println("队列已经满了。。");
            return;
        }
        arr[rear] = n;
        rear = (rear + 1) % maxSize;
    }

    //取出数据
    public int getQueue(){
        if (isEmpty()){
            throw new RuntimeException("队列为空，无法取出数据~~");
        }
        int value = arr[front];
        front = (front + 1) % maxSize;
        return value;
    }
    //显式头部数据
    public int headQueue(){
        if (isEmpty()){
            throw new RuntimeException("队列为空，无法取出数据~~");
        }
        return arr[front];
    }

    //显式队列数据
    public void showQueue(){
        if (isEmpty()){
            System.out.println("队列为空..");
            return;
        }

        for (int i = front; i < front + size();i ++){
            System.out.printf("arr[%d] = %d\n",i % maxSize, arr[i % maxSize]);
        }
    }

    //数组有效个数
    public int size(){
        return (rear + maxSize - front) % maxSize;
    }
}

    static class ArrayQueue{

    int front;//指向头部前一个位置的数据；
    int rear;//指向尾部数据


    public ArrayQueue(int arrMaxSize) {
    maxSize = arrMaxSize;
    arr = new int[maxSize];
    front = -1;
    rear = -1;
    }



    //判断队列是否为空



    public boolean isEmpty(){
    return front == rear;
    }
    //判断队列是否为满
    public boolean isFull(){
    return rear == maxSize - 1;
    }

    //向队列加入数据

    public void addQueue(int n){
    if(isFull()){
    System.out.println(“队列已经满了。。”);
    return;
    }
    rear ++;
    arr[rear] = n;
    }

    //从队列取出数据
    public int getQueue(){
    if (isEmpty()){
    throw new RuntimeException(“队列为空，无法取出数据“);
    }
    front ++;
    return arr[front];
    }

    //显式队列头数据，显示出来但是不取出

    public int headQueue(){
    if (isEmpty()){
    throw new RuntimeException(“队列为空，无法取出数据    “);
    }

    }

    System.out.println(“队列为空..”);

    for (int i = 0; i < arr.length;i ++){

package sparseArray;

/**
 * 稀疏数组与队列
 * 用于压缩数据存储空间
 */
    public class Sparsearray {

    /*
        1.创建原始二维数组；
        2.将原始数组转化为稀疏数组；
        3.将稀疏数组还原；
     */

    public static void main(String[] args) {
        //1.创建原始二维数组；
        int[][] charsArray = new int[11][11];
        charsArray[1][2] = 1;
        charsArray[2][3] = 2;
        for (int i = 0;i < 11; i ++){
            for (int j = 0; j < 11; j++) {
                System.out.print(charsArray[i][j] + "   ");
            }
            System.out.println();
        }

        System.out.println("----------------------------");
        //2.将原始数组转化为稀疏数组；
        int sum = 0;
        for (int i = 0;i < 11; i ++){
            for (int j = 0; j < 11; j++) {
                if(charsArray[i][j] != 0){
                    sum ++;
                }
            }
        }

        int[][] speararray = new int[sum + 1][3];
        speararray[0][0] = 11;
        speararray[0][1] = 11;
        speararray[0][2] = sum;

        int count = 0;
        for (int i = 0;i < 11; i ++){
            for (int j = 0; j < 11; j++) {
                if(charsArray[i][j] != 0){
                    count ++;
                    speararray[count][0] = i;
                    speararray[count][1] = j;
                    speararray[count][2] = charsArray[i][j];
                }
            }
        }


        for (int i = 0;i < speararray.length;i ++){
            for (int j = 0; j < 3; j++) {
                System.out.print(speararray[i][j] + "   ");
            }
            System.out.println();
        }
        System.out.println("----------------------------");

        //3.将稀疏数组还原；
        int[][] chararray = new int[speararray[0][0]][speararray[0][1]];
        for (int i = 1; i < speararray.length; i++) {
            chararray[speararray[i][0]][speararray[i][1]] = speararray[i][2];
        }

        for (int i = 0;i < 11; i ++){
            for (int j = 0; j < 11; j++) {
                System.out.print(chararray[i][j] + "   ");
            }
            System.out.println();
        }
    }
}
